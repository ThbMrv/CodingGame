CREATE TABLE coachs (
  id SERIAL PRIMARY KEY,
  nom VARCHAR(100) NOT NULL,
  specialite VARCHAR(100)
);

CREATE TABLE cours (
  id SERIAL PRIMARY KEY,
  nom VARCHAR(100) NOT NULL,
  description TEXT,
  coach_id INTEGER REFERENCES coachs(id) ON DELETE SET NULL
);

CREATE TABLE abonnes (
  id SERIAL PRIMARY KEY,
  nom VARCHAR(100) NOT NULL,
  email VARCHAR(100) UNIQUE NOT NULL
);

CREATE TABLE cours_abonnes (
  cours_id INTEGER REFERENCES cours(id) ON DELETE CASCADE,
  abonne_id INTEGER REFERENCES abonnes(id) ON DELETE CASCADE,
  PRIMARY KEY (cours_id, abonne_id)
);
